简易座位预订程序

C程序
C语言
发布日期:   2023-11-23
更新日期:   2023-11-23
文章字数:   1.6k
操作系统：Windows 10
编译器：TDM-GCC 10.3.0 64-bit Release
实现功能：座位预订
#include <stdio.h>
#include <stdbool.h>
#include <ctype.h>
#include <string.h>
#define LEN 14
#define SEATS 12																// 座位总数

struct planestats {																// 建立planestats模板
	int seat_id;
	bool status;
	char last[LEN];
	char first[LEN];
};

static int empty_num;															// 记录未被预订的座位数
static bool empty_status[SEATS];												// 记录每个座位的预订状态(true为已被预订, false为未被预订)

void eatline(void);																// 清空输入缓冲区
char showmenu(void);															// 显示菜单并获取用户的选择
void showlist(int n);															// 输出未被预订的座位列表
void show_ety_num(void);														// 显示未被预订的座位数
void show_ety_list(int n);														// 显示未被预订的座位列表
char *s_gets(char *st, int n);													// 从标准输入读入数据
void assign_seat(struct planestats plane[], int n);								// 预订座位
void delete_seat(struct planestats plane[], int n);								// 取消预订座位
void makelist(const struct planestats plane[], int n);							// 记录未被预订的座位数和每个座位的预订状态
void ranklist(const struct planestats plane[], int n);							// 显示按预订人姓名顺序排列的座位列表

int main(void)
{
	char choice;
	struct planestats plane[SEATS];
	for(int index=0; index<SEATS; index++)
		plane[index].status = false;

	makelist(plane, SEATS);
	while((choice=showmenu()) != 'f')
	{
		switch(choice)
		{
			case 'a':
				show_ety_num();
				break;
			case 'b':
				show_ety_list(SEATS);
				break;
			case 'c':
				ranklist(plane, SEATS);
				break;
			case 'd':
				assign_seat(plane, SEATS);
				break;
			case 'e':
				delete_seat(plane, SEATS);
				break;
		}
	}
	puts("Bye.");

	return 0;
}

/*  函数说明：显示按预订人姓名顺序排列的座位列表
	Parameters:
		*plane -结构数组指针
		n -座位总数
    Returns:
		void
	Modify:
		2023-11-23								*/
void ranklist(const struct planestats plane[], int n)
{
	int num = n-empty_num;
	int seat[num];																// 记录已被预订的座位id
	int i, j, k;
	int ret;
	
	if(num == 0)
		puts("All seats are empty.");
	else
	{
		for(i=0, j=0; i<n; i++)
		{
			if(empty_status[i])
				seat[j++] = i;
			if(j == num)
				break;
		}

		for(i=0; i<num-1; i++)
		{
			k = i;
			for(j=i+1; j<num; j++)
			{
				if((ret=strcmp(plane[seat[k]].first, plane[seat[j]].first)))	// 如果两人的first name不相同
				{
					if(ret > 0)
						k = j;
				}
				else if(strcmp(plane[seat[k]].last, plane[seat[j]].last) > 0)
					k = j;
			}
			if(k != i)
			{
				int temp = seat[k];
				seat[k] = seat[i];
				seat[i] = temp;
			}
		}
		for(int i=0; i<num; i++)
			printf("Seat %2d: %s %s\n", seat[i]+1, plane[seat[i]].first, plane[seat[i]].last);
	}
}

/*  函数说明：预订座位
	Parameters:
		*plane -结构数组指针
		n -座位总数
    Returns:
		void
	Modify:
		2023-11-23			*/
void assign_seat(struct planestats plane[], int n)
{
	int seat_num = -1;
	char ch;
	
	if(empty_num == 0)
		puts("All seats are assigned.");
	else if(empty_num == 1)														// 如果只剩一个未被预订的座位
	{
		for(int index=0; index<n; index++)
		{
			if(!empty_status[index])
			{
				seat_num = index + 1;											// 未被预订的座位编号
				break;
			}
		}
		printf("Only seat %d is available, do you want it?\n", seat_num);
		printf("Enter y for yes, others for no: ");
		do{
			ch = getchar();
		}while(isspace(ch));
		eatline();
		if(tolower(ch) != 'y')
			seat_num = 0;
	}
	else
	{
		seat_num = 0;
		do{
			puts("Which seat do you want? Choose from this list(q to quit):");
			showlist(n);
			if(scanf("%d", &seat_num)==0 && (ch=tolower(getchar()))=='q')
			{
				eatline();
				seat_num = 0;
				break;
			}
			eatline();
		}while(seat_num<1 || seat_num>SEATS || empty_status[seat_num-1]);
	}
	
	if(seat_num > 0)
	{
		do{
			puts("Enter first name");
		}while(s_gets(plane[seat_num-1].first, LEN)==NULL || plane[seat_num-1].first[0]=='\0');
		do{
			puts("Enter last name");
		}while(s_gets(plane[seat_num-1].last, LEN)==NULL || plane[seat_num-1].last[0]=='\0');
		printf("%s %s assigned to seat %d\n", plane[seat_num-1].first, plane[seat_num-1].last, seat_num);
		puts("Enter a to accept assignment, c to cancel it.");
		while(strchr("ac", (ch=tolower(getchar()))) == NULL)
		{
			if(isspace(ch))
				continue;
			eatline();
			puts("Enter a character in the list ac");
		}
		eatline();
		if(ch == 'a')
		{
			puts("Passenger assigned to seat.");
			plane[seat_num-1].status = true;
			makelist(plane, SEATS);
		}
		else
			puts("Passenger not assigned.");
	}
	else if(seat_num == 0)
		puts("Booking failed.");
}

/*  函数说明：取消预订座位
	Parameters:
		*plane -结构数组指针
		n -座位总数
    Returns:
		void
	Modify:
		2023-11-23			*/
void delete_seat(struct planestats plane[], int n)
{
	int index, seat_num = -1;
	char ch;
	int num = n-empty_num;
	int seat[num];																// 记录已被预订的座位id
	for(int j=0, index=0; index<n; index++)
	{
		if(empty_status[index])
			seat[j++] = index;
		if(j == num)
			break;
	}
	
	if(empty_num == n)
		puts("All seats already are empty.");
	else
	{
		seat_num = 0;
		puts("Seats currently taken:");
		for(index=0; index<num; index++)
				printf("Seat %d: %s %s\n", seat[index]+1, plane[seat[index]].first, plane[seat[index]].last);
		puts("Enter the number of the seat to be cancelled(q to quit):");
		while(scanf("%d", &seat_num)==1 || (ch=tolower(getchar()))!='q')
		{
			if(seat_num>=1 && seat_num<=SEATS && empty_status[seat_num-1])
				break;
			eatline();
			seat_num = 0;
			puts("Enter a number from this list:");
			for(index=0; index<num; index++)
				printf("%d ", seat[index]+1);
			putchar('\n');
		}
		eatline();
	}
	
	if(seat_num > 0)
	{
		printf("%s %s to be canceled for seat %d.\n", plane[seat_num-1].first, plane[seat_num-1].last, seat_num);
		puts("Enter d to delete assignment, a to abort.");
		while(!strchr("ad", (ch=tolower(getchar()))))
		{
			if(isspace(ch))
				continue;
			eatline();
			puts("Enter a character in the list ad");
		}
		eatline();
		if(ch == 'd')
		{
			puts("Passenger dropped.");
			plane[seat_num-1].status = false;
			makelist(plane, SEATS);
		}
		else
			puts("Passenger retained.");
	}
	else if(seat_num == 0)
		puts("Unsubscribe failed.");
}

/*  函数说明：显示未被预订的座位列表
    Parameters:
		n -座位总数
	Returns:
		void
	Modify:
		2023-11-23					*/
void show_ety_list(int n)
{
	if(empty_num == 0)
		puts("All seats are assigned.");
	else
	{
		if(empty_num == 1)
			puts("The following seat is available.");
		else
			puts("The following seats are available.");
		showlist(n);
	}
}

/*  函数说明；显示未被预订的座位数
	Parameters:
		void
    Returns:
		void
	Modify:
		2023-11-23					*/
void show_ety_num(void)
{
	if(empty_num == 1)
		puts("There is 1 empty seat.");
	else
		printf("There are %d empty seats.\n", empty_num);
}

/*  函数说明；记录未被预订的座位数和每个座位的预订状态
	Parameters:
		*plane -结构数组指针
		n -座位总数
    Returns:
		void
	Modify:
		2023-11-23										*/
void makelist(const struct planestats plane[], int n)
{
	int index;

	empty_num = 0;
	for(index=0; index<n; index++)
	{
		empty_status[index] = plane[index].status;
		if(!empty_status[index])
			empty_num++;
	}
}

/*  函数说明：输出未被预订的座位列表
    Parameters:
		n -座位总数
	Returns:
		void
	Modify:
		2023-11-23					*/
void showlist(int n)
{
	int index, ipnum = 0;
	
	for(index=0; index<n; index++)
	{
		if(!empty_status[index])
		{
			printf("%d ", index+1);
			ipnum++;
			if(ipnum == empty_num)
				break;
		}
	}
	putchar('\n');
}

/* 	函数说明：显示菜单并获取用户的选择
    Parameters:
		void
	Returns:
		ch -用户的选择
	Modify:
		2023-11-23						*/
char showmenu(void)
{
	char ch;

	printf("To choose a function, enter its letter label:\n"
		   "a) Show number of empty seats\n"
		   "b) Show list of empty seats\n"
		   "c) Show alphabetical list of seats\n"
		   "d) Assign a customer to a seat assignment\n"
		   "e) Delete a seat assignment\n"
		   "f) Quit\n");
	while(strchr("abcdef", ch=tolower(getchar())) == NULL)
	{
		if(isspace(ch))
			continue;
		eatline();
		puts("Enter a character in the list abcdef");
	}
	eatline();

	return ch;
}

/*  函数说明：从标准输入读入数据
    Parameters:
		*st -字符串数组指针
		n -读入字符的最大数量
    Returns:
		*ret_val -fgets()函数返回地址
    Modify:
		2023-11-20						*/
char *s_gets(char *st, int n)
{
	char *ret_val;
	char *find;

	ret_val = fgets(st, n, stdin);
	if(ret_val)
	{
		find = strchr(st, '\n');												// 查找换行符
		if(find)																// 如果地址不是NULL
			*find = '\0';														// 在此处放置一个空字符
        else
            eatline();															// 处理输入行中剩余的字符
	}

	return ret_val;
}

/* 	函数说明：清空输入缓冲区
    Parameters:
		void
	Returns:
		void
	Modify:
		2023-11-20          */
void eatline(void)
{
	while(getchar() != '\n')
		continue;
}
