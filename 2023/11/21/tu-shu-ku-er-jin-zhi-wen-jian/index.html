图书库二进制文件

C程序
C语言
发布日期:   2023-11-21
更新日期:   2023-11-21
文章字数:   2k
操作系统：Windows 10
编译器：TDM-GCC 10.3.0 64-bit Release
实现功能：从二进制文件中读取每条记录并显示出来，允许用户新增记录、删除记录或修改记录的内容
#include <stdio.h>
#include <stdlib.h>
#include <stdbool.h>
#include <string.h>
#define MAXTITL 40																// 书名最大长度
#define MAXAUTL 40																// 作者名最大长度
#define MAXBKS 10																// 图书的最大数量

struct book {																	// 建立book模板
	char title[MAXTITL];
	char author[MAXAUTL];
	float value;
};

struct pack {																	// 增加删除和修改标志位
	struct book book;
	bool delete_me;
	bool revise_me;
};

static int size = sizeof(struct book);											// book模板字节数
static char filename[] = "book.dat";											// 图书库文件名称

void eatline(void);																// 清空输入缓冲区
int showmenu(void);																// 显示菜单并获取用户的选择
char *s_gets(char *st, int n);													// 从标准输入读入数据
int delrecord(struct pack library[], int filecount);							// 删除图书记录
int addrecord(struct pack library[], int filecount);							// 新增图书记录
void modrecord(struct pack library[], int filecount);							// 修改图书记录
void showrecord(const struct pack library[], int filecount);					// 输出当前结构数组中的图书记录
void wirterecord(FILE *pbooks, const struct pack library[], int filecount);		// 重写图书库文件内容
void reviserecord(FILE *pbooks, const struct pack library[], int filecount);	// 按需修改图书库文件内容

int main(void)
{
	struct pack library[MAXBKS];												// 结构数组
	int initial_count = 0;														// 记录原始图书数量
	int count = 0;																// 记录当前图书数量
	int choice;
	FILE *pbooks;

	if((pbooks=fopen(filename, "r+b")) == NULL)
	{
		fprintf(stderr, "Can't open %s file.\n", filename);
		exit(EXIT_FAILURE);
	}
	rewind(pbooks);																// 定位到文件开始
	while(count<MAXBKS && fread(&library[count].book, size, 1, pbooks) == 1)
	{
		if(count == 0)
			printf("Current contents of %s:\n", filename);
		printf("%d.%s by %s: $%.2f\n", count+1, library[count].book.title, library[count].book.author, library[count].book.value);
		library[count].delete_me = false;										// 将删除标志位置为false
		library[count].revise_me = false;										// 将修改标志位置为false
		count++;
	}
	initial_count = count;
	if(count == 0)
		puts("No books? Too bad.");
		
	while((choice=showmenu()) != 4)												// 当用户的选择不是4时
	{
		switch(choice)
		{
			case 1:
				count = addrecord(library, count);								// 新增图书记录
				showrecord(library, count);
				break;
			case 2:
				count = delrecord(library, count);								// 删除图书记录
				showrecord(library, count);
				break;
			case 3:
				modrecord(library, count);										// 修改图书记录
				break;
		}
	}
	
	if(count<initial_count)														// 如果当前图书数量小于原始图书数量, 则重写图书库文件
		wirterecord(pbooks, library, count);
	else																		// 否则, 按需修改图书库文件内容
		reviserecord(pbooks, library, count);
	puts("Bye.");

	return 0;
}

/*  函数说明：修改图书记录
    Parameters:
		*library -结构数组指针
		filecount -当前图书数量
    Returns:
		void
	Modify:
		2023-11-20						*/
void modrecord(struct pack library[], int filecount)
{
	int id, index;
	int skipt;
	bool modify;
	
	if(filecount)
	{
		printf("请输入你要修改的记录id(按q退出): ");
		while(scanf("%d", &id) == 1)
		{
			if(id<1 || id>filecount)
			{
				eatline();
				printf("没有编号为%d的记录.\n", id);
				printf("请输入其他要修改的记录id(按q退出): ");
				continue;
			}
			
			eatline();
			skipt = index = 0;
			do{																	// 找到记录id对应的下标
				if(!library[index++].delete_me)
					skipt++;
			}while(skipt<id);
			
			modify = false;
			printf("修改金额请按1, 修改书名请按2, 修改作者请按3, 按其他键放弃修改: ");
			if(scanf("%d", &id) == 1)
			{
				eatline();
				switch(id)
				{
					case 1:
						puts("输入修改后的金额, 非数字键退出.");
						if(scanf("%f", &library[index-1].book.value) == 1)
							modify = true;
						eatline();
						break;
					case 2:
						{
							char temp[MAXTITL];
							puts("输入修改后的书名, 空行退出.");
							if(s_gets(temp, MAXTITL)!=NULL && temp[0]!='\0')
							{
								strcpy(library[index-1].book.title, temp);
								modify = true;
							}
						}
						break;
					case 3:
						{
							char temp[MAXAUTL];
							puts("输入修改后的作者名, 空行退出.");
							if(s_gets(temp, MAXTITL)!=NULL && temp[0]!='\0')
							{
								strcpy(library[index-1].book.author, temp);
								modify = true;
							}
						}
						break;
				}
			}
			else
				eatline();
			
			if(modify)
			{
				library[index-1].revise_me = true;
				puts("Successfully modified.");
			}
			else
				puts("Modification failed.");
			showrecord(library, filecount);
			printf("请输入其他要修改的记录id(按q退出): ");
		}
		eatline();
	}
	else
		puts("No books? Too bad.");
}

/*  函数说明：删除图书记录
    Parameters:
		*library -结构数组指针
		filecount -当前图书数量
    Returns:
		count -进行删除操作后剩余的图书数量
	Modify:
		2023-11-20              			*/
int delrecord(struct pack library[], int filecount)
{
	int id, index, count = filecount;
	int skipt;
	bool delme[filecount];														// 删除标志位数组
	for(int i=0; i<filecount; i++)
		delme[i] = false;
		
	if(filecount)
	{
		printf("请输入你要删除的记录id(按q退出): ");
		while(scanf("%d", &id) == 1)
		{
			eatline();
			if(id<1 || id>filecount)
			{
				printf("没有编号为%d的记录.\n", id);
				printf("请输入其他要删除的记录id(按q退出): ");
				continue;
			}
			
			if(delme[id-1])
			{
				printf("编号为%d的记录已删除.\n", id);
			}
			else
			{
				skipt = index = 0;
				do{																// 找到记录id对应的下标
					if(!library[index++].delete_me)
						skipt++;
				}while(skipt<id);
				library[index-1].delete_me = true;
				delme[id-1] = true;												// 将删除标志位置为true
				if(--count == 0)
					break;
			}
			printf("请输入其他要删除的记录id(按q退出): ");
		}
		eatline();
	}
	
	return count;
}

/*  函数说明：新增图书记录
    Parameters:
		*library -结构数组指针
		filecount -当前图书数量
    Returns:
		count -进行新增图书操作后的图书数量
	Modify:
		2023-11-20              			*/
int addrecord(struct pack library[], int filecount)
{
	int id = 0;
	
	if(filecount == MAXBKS)
	{
		fprintf(stderr, "The %s file is full.\n", filename);
	}
	else
	{
		puts("Please add new book titles.");
		puts("Press [enter] at the start of a line to stop.");
		while(id < MAXBKS)
		{
			if(((id<filecount && library[id].delete_me) || (id == filecount)))	// 判断当前记录是否被删除或为空记录
			{
				if(s_gets(library[id].book.title, MAXTITL)!=NULL && library[id].book.title[0]!='\0')
				{
					printf("Now enter the author.\n");
					s_gets(library[id].book.author, MAXAUTL);
					printf("Now enter the value.\n");
					scanf("%f", &library[id].book.value);
					eatline();
					library[id].delete_me = false;								// 将删除标志位置为false
					library[id++].revise_me = true;								// 将修改标志位置为true
					filecount++;
				}
				else
					break;
			}
			else
			{
				id++;
				continue;
			}
			if(filecount < MAXBKS)
				puts("Enter the next title.");
			else
  			{
				printf("The %s file is full.\n", filename);
			}
  		}
	}
	
	return filecount;
}

/*  函数说明：按需修改图书库文件内容
    Parameters:
		*pbooks -图书库文件指针
		*library -结构数组指针
		filecount -当前图书数量
    Returns:
		void
	Modify:
		2023-11-20              	*/
void reviserecord(FILE *pbooks, const struct pack library[], int filecount)
{
	int count = 0, id = 0;
	long pos;

	while(id<MAXBKS && count<filecount)
	{
		if(!library[id].delete_me)
		{
			if(library[id].revise_me)
			{
				pos = (long)id * size;											// 计算偏移量
				fseek(pbooks, pos, SEEK_SET);									// 定位到此处
				fwrite(&library[id].book, size, 1, pbooks);
			}
			count++;
		}
		id++;
	}
	if(fclose(pbooks) != 0)
		fprintf(stderr, "Can't close file %s.\n", filename);
}

/*  函数说明：重写图书库文件
    Parameters:
		*pbooks -图书库文件指针
		*library -结构数组指针
		filecount -当前图书数量
    Returns:
		void
	Modify:
		2023-11-20              */
void wirterecord(FILE *pbooks, const struct pack library[], int filecount)
{
	int count = 0, id = 0;
	
	if(fclose(pbooks) != 0)
	{
		fprintf(stderr, "Can't close file %s.\n", filename);
		exit(EXIT_FAILURE);
	}
	if((pbooks=fopen("book.dat", "wb")) == NULL)
	{
		fprintf(stderr, "Can't revise file %s.\n", filename);
		exit(EXIT_FAILURE);
	}
	
	rewind(pbooks);																// 定位到文件开始
	while(id<MAXBKS && count<filecount)
	{
		if(!library[id].delete_me)
		{
			fwrite(&library[id].book, size, 1, pbooks);
			count++;
		}
		id++;
	}
	fclose(pbooks);
}

/*	函数说明：输出当前结构数组中的图书记录
    Parameters:
		*library -结构数组指针
		filecount -当前图书数量
    Returns:
		void
	Modify:
		2023-11-20              			*/
void showrecord(const struct pack library[], int filecount)
{
	int count = 0, id;

	if(filecount)
	{
		for(id=0; id<MAXBKS; id++)
		{
			if(id == 0)
				printf("Current contents of %s:\n", filename);
			if(!library[id].delete_me)
				printf("%d.%s by %s: $%.2f\n", ++count, library[id].book.title, library[id].book.author, library[id].book.value);
			if(count == filecount)
				break;
		}
	}
	else
		puts("No books? Too bad.");
}

/* 	函数说明：显示菜单并获取用户的选择
    Parameters:
		void
	Returns:
		ret -用户的选择
	Modify:
		2023-11-20						*/
int showmenu(void)
{
	int ret = 0;

	printf("1) 新增记录  2) 删除记录\n"
		   "3) 修改记录  4) 退出\n");
	printf("请输入您的选择: ");
	scanf("%d", &ret);
	eatline();
	while(ret<1 || ret>4)
	{
		printf("请输入1~4的正整数: ");
		scanf("%d", &ret);
		eatline();
	}

	return ret;
}

/*  函数说明：从标准输入读入数据
    Parameters:
		*st -字符串数组指针
		n -读入字符的最大数量
    Returns:
		*ret_val -fgets()函数返回地址
    Modify:
		2023-11-20                      */
char *s_gets(char *st, int n)
{
	char *ret_val;
	char *find;

	ret_val = fgets(st, n, stdin);
	if(ret_val)
	{
		find = strchr(st, '\n');												// 查找换行符
		if(find)																// 如果地址不是NULL
			*find = '\0';														// 在此处放置一个空字符
        else
            eatline();															// 处理输入行中剩余的字符
	}

	return ret_val;
}

/* 	函数说明：清空输入缓冲区
    Parameters:
		void
	Returns:
		void
	Modify:
		2023-11-20          */
void eatline(void)
{
	while(getchar() != '\n')
		continue;
}
